==== Deployment

ssh -i "pem_file" ubuntu@ec2-52-26-134-51.us-west-2.compute.amazonaws.com

cd images-store

git pull origin master

bundle install

rbenv rehash

rake db:migrate RAILS_ENV=production
    rake assets:precompile RAILS_ENV=production

rbenv sudo passenger stop -p80

rbenv sudo passenger start -p80 -d -e production

connect at: ec2-52-26-134-51.us-west-2.compute.amazonaws.com
